    <div>

      <button onClick={() => actions.add()}>add</button>
    </div>
  );
}


function Counter() {
  const [state, actions] = useModel(countModel);

  return (
    <div>
      <div>counter: {state.value}</div>
      <button onClick={() => actions.add()}>add</button>
    </div>
  );
}

export default function App() {
  return <LocalCounterProvider>
    <Counter />
    <LocalCounter />
  </LocalCounterProvider>;
}
